import math
def cubo(a, b, c):
	if (a == b) & (a == c):
		
		cubo = a*b*c
		return cubo
	else:
		return 0

def piramideTriangular(l,h):
	VolumenPiramide = (math.sqrt(3)*(l**2)*h)
	return VolumenPiramide

def piramideCuadrangular(l,h):
	VolumenPiramideCuadrangular = (((l**2)*h)/3)
	return VolumenPiramideCuadrangular
